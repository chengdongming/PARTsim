## 🔧 代码实现细节

### 能量预算检查代码位置

**文件**: [librtsim/scheduler/gpfp_asap_scheduler.cpp](../librtsim/scheduler/gpfp_asap_scheduler.cpp)

**关键代码段**: 第1995-2056行

```cpp
} else if (current_energy >= unit_energy - 1e-6) {
    // 计算队列中已有任务的预估总能耗
    double estimated_batch_energy = unit_energy;  // 当前任务

    size_t queue_size = getSize();
    size_t check_count = std::min(queue_size, static_cast<size_t>(_num_cores));

    for (size_t i = 0; i < check_count; ++i) {
        AbsRTTask* queued_task = getTaskN(i);
        if (queued_task) {
            double queued_energy = getUnitTimeEnergy(queued_task);
            estimated_batch_energy += queued_energy;
        }
    }

    // 如果预估能耗超过当前能量,需要更严格的优先级检查
    if (estimated_batch_energy > current_energy + 1e-6) {
        // 能量紧张,只允许高优先级任务进入队列
        bool should_insert = false;
        int current_prio = getPriority(task);

        // 检查就绪队列中是否有低优先级任务
        for (size_t i = 0; i < queue_size; ++i) {
            AbsRTTask* queued_task = getTaskN(i);
            if (queued_task) {
                int queued_prio = getPriority(queued_task);

                if (current_prio < queued_prio) {
                    // 当前任务优先级更高,应该插入
                    should_insert = true;
                    break;
                }
            }
        }

        if (!should_insert) {
            // 当前任务优先级不够高,加入等待队列
            _waiting_queue.push_back(task);
            _stats.total_skipped_energy++;
            return;
        }
    }
}
```
